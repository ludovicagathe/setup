alias gcts="read input_var && echo \"$(date +%Y%m%d%H%M)\" - $input_var"

# FUNCTIONS
# git commit -am "YYYYmmddHHMM - Details"
gcts() {
  if [[ -z $1 ]]; then
    DETAILS="Work in progress"
  else
    DETAILS=$1
  fi
  git commit -am "$(date +%Y%m%d%H%M) - $DETAILS"
}
